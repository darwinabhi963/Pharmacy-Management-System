CREATE DATABASE ipproject;
USE ipproject;
CREATE TABLE USERS(USERNAME VARCHAR(20) PRIMARY KEY,PASSWORD VARCHAR(20) UNIQUE NOT NULL,DEPT VARCHAR(20) NOT NULL,EMAILID VARCHAR(20),MOBILENO LONG);
CREATE TABLE ADMIN(ADMINNO INTEGER(20) PRIMARY KEY,ADMINNAME VARCHAR(20) NOT NULL,ADMINAGE INTEGER(5) NOT NULL,ADMINGENDER VARCHAR(10),ADMINDEPT VARCHAR(20) NOT NULL,ADMINMOB LONG NOT NULL,ADMINADD VARCHAR(300),ADMINMAIL VARCHAR(50) NOT NULL);
CREATE TABLE CUSTOMER(CUSNAME VARCHAR(20) NOT NULL,CUSAGE INTEGER(5) NOT NULL,CUSGENDER VARCHAR(10) NOT NULL,CUSMOB LONG NOT NULL,CUSADD VARCHAR(300),CUSMAIL VARCHAR(50) NOT NULL);
CREATE TABLE MEDICINES(MEDID INTEGER(20) PRIMARY KEY,MEDNAME VARCHAR(30) UNIQUE NOT NULL,MEDFUNC VARCHAR(50) NOT NULL,MEDCOMP VARCHAR(30) NOT NULL,MEDPRICE NUMERIC(10,2) NOT NULL,MEDMFD DATE NOT NULL,MEDEXP DATE NOT NULL,MEDMANF VARCHAR(40) NOT NULL,MEDDOSE VARCHAR(20) NOT NULL,SCHEDULEDHDRUG VARCHAR(5) NOT NULL);
CREATE TABLE ORDERS(ORDERID INTEGER(20) PRIMARY KEY,CUSNAME VARCHAR(50) NOT NULL,ORDERPRICE NUMERIC(10,2) NOT NULL,PLACEDATE DATE NOT NULL,DELIVERYDATE DATE NOT NULL,NOOFITEMS INTEGER(225) NOT NULL,PAYMODE VARCHAR(50) NOT NULL);
CREATE TABLE CART(
    -> SNO INTEGER(10) PRIMARY KEY,
    -> MEDNAME VARCHAR(50) NOT NULL,
    -> MEDCOMP VARCHAR(50) NOT NULL,
    -> MEDPRICE NUMERIC(10,2) NOT NULL,
    -> MEDMFD DATE NOT NULL,
    -> MEDEXP DATE NOT NULL,
    -> MEDMANF VARCHAR(50) NOT NULL,
    -> MEDDOSE VARCHAR(10) NOT NULL,
    -> SCHEDULEDHDRUG VARCHAR(10) NOT NULL,
    -> QUANTITY INTEGER(10) NOT NULL);
CREATE TABLE FEEDBACK(CUSNAME VARCHAR(20) NOT NULL,FEEDBACK VARCHAR(100));
CREATE TABLE NETBANKING(USERNAME VARCHAR(20),PASSWORD VARCHAR(20),ACCOUNTNO LONG);
CREATE TABLE CREDITCARD(EXPMM INTEGER(2),EXPYYYY INTEGER(4),CVV INTEGER(3),CARDNO LONG);
CREATE TABLE DEBITCARD(EXPMM INTEGER(2),EXPYYYY INTEGER(4),CVV INTEGER(3),CARDNO LONG);
